%% Information sheet — a single-page limits table over every problem.
%%
%% NOTE (statements v2): an info sheet needs per-problem data (limits), and the
%% `documents` section never receives the `problems` list. So this is modelled
%% as an rbx contest *statement* that joins on problem *metadata*: it iterates
%% `problems` for the table but does NOT `\subimport` their bodies. Its
%% `contestProblemTemplate` (`info-fragment.rbx.tex`) is therefore intentionally
%% empty, and it carries no `standaloneProblemTemplate` so it is never a
%% standalone candidate.
\documentclass[a4paper,11pt]{article}

\def\lang{\VAR{lang}}
\usepackage{icpc}

\markright{\VAR{contest.title}}

\begin{document}

\begin{titlepage}
\vspace*{\fill}
\begin{center}
\includegraphics[width=4cm]{logo.png}

\vspace{1cm}
{\huge{\bf \VAR{vars.year}}} \\

\vspace{0.5cm}
{\huge \bf \strInfoSheet}\\[12pt]
{\Large \bf \VAR{contest.title}}
\end{center}

\vspace{2cm}

\begin{center}
\begin{tabular}{c|ccc|c}
& \multicolumn{3}{c|}{{\strTimeLimit} (ms)} & \multirow{2}{*}{{\strMemoryLimit} (MiB)} \\
{\sf \strProblem} & {\sf C/C++} &{\sf Java} & {\sf Python3} & \\
\hline
%- for problem in problems
\VAR{problem.short_name}
& \VAR{problem.limits.timelimit_for_language('cpp')}
& \VAR{problem.limits.timelimit_for_language('java')}
& \VAR{problem.limits.timelimit_for_language('py')}
& \VAR{problem.limits.memorylimit_for_language()}
\\ \hline
%- endfor
\end{tabular}
\end{center}
\vspace*{\fill}
\end{titlepage}
\end{document}
